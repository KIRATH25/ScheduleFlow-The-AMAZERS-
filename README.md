```javascript
function setupFeaturesCarousel() {
    const carousel = document.getElementById('featuresCarousel');
    
    // Create infinite scroll by cloning cards
    const originalCards = carousel.querySelectorAll('.feature-card-carousel');
    originalCards.forEach(card => {
        const clone = card.cloneNode(true);
        carousel.appendChild(clone);
    });
    
    // Auto-scroll with pause on hover
    carousel.addEventListener('mouseenter', () => {
        carousel.style.animationPlayState = 'paused';
    });
    
    carousel.addEventListener('mouseleave', () => {
        carousel.style.animationPlayState = 'running';
    });
}
```

        video.className = 'hero-video';
        video.autoplay = true;
        video.muted = true;
        video.loop = true;
        if (posterSrc) video.poster = posterSrc;
        
        const source = document.createElement('source');
        source.src = videoSrc;
        source.type = 'video/mp4';
        video.appendChild(source);
        
        insertVideoElement(video);
        return video;
    },

        const iframe = document.createElement('iframe');
        iframe.className = 'hero-video';
        iframe.src = `https://www.youtube.com/embed/${videoId}?autoplay=1&mute=1&loop=1`;
        iframe.frameBorder = '0';
